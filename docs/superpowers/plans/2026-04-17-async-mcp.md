# 全系统异步化 + MCP 协议栈 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将 viv 的 Agent 线程从同步阻塞改为全异步，并实现完整的 MCP 客户端（stdio/SSE/HTTP/WebSocket 传输，tools/resources/prompts）。

**Architecture:** UI 线程保持同步 epoll 驱动不变。Agent 侧全部运行在自制 async runtime (executor + reactor) 上。MCP 客户端通过 Transport trait 抽象四种传输，McpManager 管理多服务器连接，动态注册 MCP 工具到现有 ToolRegistry。

**Tech Stack:** Rust edition 2024, 零外部依赖, 自制 async runtime (epoll), OpenSSL FFI

**Spec:** `docs/superpowers/specs/2026-04-17-async-mcp-design.md`

创建 `src/core/jsonrpc.rs`：

```rust
use crate::core::json::{JsonValue, Number};

// ── Request ──────────────────────────────────────────────────────────────────

pub struct Request {
    pub id: i64,
    pub method: String,
    pub params: Option<JsonValue>,
}

impl Request {
    pub fn to_json(&self) -> JsonValue {
        let mut pairs = vec![
            ("jsonrpc".into(), JsonValue::Str("2.0".into())),
            ("id".into(), JsonValue::Number(Number::Int(self.id))),
            ("method".into(), JsonValue::Str(self.method.clone())),
        ];
        if let Some(ref p) = self.params {
            pairs.push(("params".into(), p.clone()));
        }
        JsonValue::Object(pairs)
    }
}

// ── Response ─────────────────────────────────────────────────────────────────

pub enum Response {
    Result { id: i64, result: JsonValue },
    Error { id: i64, error: RpcError },
}

pub struct RpcError {
    pub code: i64,
    pub message: String,
    pub data: Option<JsonValue>,
}

// ── Notification ─────────────────────────────────────────────────────────────

pub struct Notification {
    pub method: String,
    pub params: Option<JsonValue>,
}

impl Notification {
    pub fn to_json(&self) -> JsonValue {
        let mut pairs = vec![
            ("jsonrpc".into(), JsonValue::Str("2.0".into())),
            ("method".into(), JsonValue::Str(self.method.clone())),
        ];
        if let Some(ref p) = self.params {
            pairs.push(("params".into(), p.clone()));
        }
        JsonValue::Object(pairs)
    }
}

// ── Message ──────────────────────────────────────────────────────────────────

pub enum Message {
    Request(Request),
    Response(Response),
    Notification(Notification),
}

impl Message {
    pub fn parse(json: &JsonValue) -> crate::Result<Self> {
        // 有 id + method → Request
        // 有 id + result/error → Response
        // 有 method 无 id → Notification
        let has_id = json.get("id").is_some();
        let has_method = json.get("method").is_some();
        let has_result = json.get("result").is_some();
        let has_error = json.get("error").is_some();

        if has_id && has_method {
            let id = json.get("id").and_then(|v| v.as_i64())
                .ok_or(crate::Error::Json("JSON-RPC: id must be integer".into()))?;
            let method = json.get("method").and_then(|v| v.as_str())
                .ok_or(crate::Error::Json("JSON-RPC: method must be string".into()))?
                .to_string();
            let params = json.get("params").cloned();
            return Ok(Message::Request(Request { id, method, params }));
        }

        if has_id && has_result {
            let id = json.get("id").and_then(|v| v.as_i64())
                .ok_or(crate::Error::Json("JSON-RPC: id must be integer".into()))?;
            let result = json.get("result").cloned()
                .ok_or(crate::Error::Json("JSON-RPC: missing result".into()))?;
            return Ok(Message::Response(Response::Result { id, result }));
        }

        if has_id && has_error {
            let id = json.get("id").and_then(|v| v.as_i64())
                .ok_or(crate::Error::Json("JSON-RPC: id must be integer".into()))?;
            let err_obj = json.get("error")
                .ok_or(crate::Error::Json("JSON-RPC: missing error".into()))?;
            let code = err_obj.get("code").and_then(|v| v.as_i64())
                .ok_or(crate::Error::Json("JSON-RPC: error.code must be integer".into()))?;
            let message = err_obj.get("message").and_then(|v| v.as_str())
                .ok_or(crate::Error::Json("JSON-RPC: error.message must be string".into()))?
                .to_string();
            let data = err_obj.get("data").cloned();
            return Ok(Message::Response(Response::Error {
                id,
                error: RpcError { code, message, data },
            }));
        }

        if has_method {
            let method = json.get("method").and_then(|v| v.as_str())
                .ok_or(crate::Error::Json("JSON-RPC: method must be string".into()))?
                .to_string();
            let params = json.get("params").cloned();
            return Ok(Message::Notification(Notification { method, params }));
        }

        Err(crate::Error::Json("JSON-RPC: unrecognized message format".into()))
    }
}
```

创建 `src/core/runtime/channel.rs`：

```rust
use std::sync::mpsc;
use std::os::unix::io::RawFd;
use std::task::{Context, Poll};
use std::pin::Pin;
use std::future::Future;
use super::reactor::reactor;

// pipe FFI
unsafe extern "C" {
    fn pipe(pipefd: *mut [RawFd; 2]) -> i32;
    fn close(fd: RawFd) -> i32;
    fn write(fd: RawFd, buf: *const u8, count: usize) -> isize;
    fn read(fd: RawFd, buf: *mut u8, count: usize) -> isize;
}

// fcntl for non-blocking
unsafe extern "C" {
    fn fcntl(fd: RawFd, cmd: i32, ...) -> i32;
}
const F_GETFL: i32 = 3;
const F_SETFL: i32 = 4;
const O_NONBLOCK: i32 = 0o4000;

fn set_nonblocking(fd: RawFd) {
    unsafe {
        let flags = fcntl(fd, F_GETFL);
        fcntl(fd, F_SETFL, flags | O_NONBLOCK);
    }
}

pub struct NotifySender<T> {
    tx: mpsc::Sender<T>,
    pipe_write: RawFd,
}

impl<T> NotifySender<T> {
    pub fn send(&self, value: T) -> Result<(), mpsc::SendError<T>> {
        self.tx.send(value)?;
        // 写 1 字节唤醒 reactor
        unsafe { write(self.pipe_write, &1u8 as *const u8, 1); }
        Ok(())
    }
}

impl<T> Clone for NotifySender<T> {
    fn clone(&self) -> Self {
        NotifySender {
            tx: self.tx.clone(),
            pipe_write: self.pipe_write,
        }
    }
}

pub struct AsyncReceiver<T> {
    rx: mpsc::Receiver<T>,
    pipe_read: RawFd,
    token: Option<u64>,
}

impl<T> AsyncReceiver<T> {
    pub fn recv(&self) -> RecvFuture<'_, T> {
        RecvFuture { receiver: self, token: None }
    }
}

pub struct RecvFuture<'a, T> {
    receiver: &'a AsyncReceiver<T>,
    token: Option<u64>,
}

impl<T> Future for RecvFuture<'_, T> {
    type Output = crate::Result<T>;

    fn poll(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
        // 先排干 pipe 中的通知字节
        let mut drain = [0u8; 64];
        unsafe { read(self.receiver.pipe_read, drain.as_mut_ptr(), drain.len()); }

        // 尝试接收
        match self.receiver.rx.try_recv() {
            Ok(value) => {
                // 清除旧 token
                if let Some(t) = self.token.take() {
                    reactor().lock().unwrap().remove(t);
                }
                Poll::Ready(Ok(value))
            }
            Err(mpsc::TryRecvError::Empty) => {
                // 注册 pipe_read 到 reactor
                if let Some(t) = self.token.take() {
                    reactor().lock().unwrap().remove(t);
                }
                let t = reactor().lock().unwrap()
                    .register_readable(self.receiver.pipe_read, cx.waker().clone());
                self.token = Some(t);
                Poll::Pending
            }
            Err(mpsc::TryRecvError::Disconnected) => {
                Poll::Ready(Err(crate::Error::Io(
                    std::io::Error::new(std::io::ErrorKind::BrokenPipe, "channel closed")
                )))
            }
        }
    }
}

impl<T> Drop for RecvFuture<'_, T> {
    fn drop(&mut self) {
        if let Some(t) = self.token.take() {
            reactor().lock().unwrap().remove(t);
        }
    }
}

pub fn async_channel<T>() -> (NotifySender<T>, AsyncReceiver<T>) {
    let (tx, rx) = mpsc::channel();
    let mut fds = [0i32; 2];
    unsafe { pipe(&mut fds as *mut [RawFd; 2]); }
    set_nonblocking(fds[0]);

    (
        NotifySender { tx, pipe_write: fds[1] },
        AsyncReceiver { rx, pipe_read: fds[0], token: None },
    )
}
```

```rust
use super::async_tcp::AsyncTcpStream;
use crate::core::runtime::reactor::reactor;

// OpenSSL FFI（复用 tls.rs 的声明方式）
unsafe extern "C" {

    fn TLS_client_method() -> *mut c_void;

    fn SSL_CTX_set_default_verify_paths(ctx: *mut c_void) -> i32;

    fn SSL_ctrl(ssl: *mut c_void, cmd: i32, larg: i64, parg: *mut c_void) -> i64;

    fn SSL_get_error(ssl: *mut c_void, ret: i32) -> i32;
}

const SSL_VERIFY_PEER: i32 = 0x01;
const SSL_ERROR_WANT_READ: i32 = 2;
const SSL_ERROR_WANT_WRITE: i32 = 3;

pub struct AsyncTlsStream {
    tcp: AsyncTcpStream,
    ssl: *mut c_void,
    ctx: *mut c_void,
}

unsafe impl Send for AsyncTlsStream {}

impl AsyncTlsStream {
    pub async fn connect(host: &str, port: u16) -> crate::Result<Self> {
        let tcp = AsyncTcpStream::connect(host, port).await?;
        let fd = tcp.raw_fd();

        // OpenSSL init + context
        unsafe { OPENSSL_init_ssl(0, std::ptr::null()); }
        let method = unsafe { TLS_client_method() };
        if method.is_null() {
            return Err(crate::Error::Tls("TLS_client_method failed".into()));
        }
        let ctx = unsafe { SSL_CTX_new(method) };
        if ctx.is_null() {
            return Err(crate::Error::Tls("SSL_CTX_new failed".into()));
        }
        unsafe {
            SSL_CTX_set_default_verify_paths(ctx);
            SSL_CTX_set_verify(ctx, SSL_VERIFY_PEER, std::ptr::null());
        }

        let ssl = unsafe { SSL_new(ctx) };
        if ssl.is_null() {
            unsafe { SSL_CTX_free(ctx); }
            return Err(crate::Error::Tls("SSL_new failed".into()));
        }
        if unsafe { SSL_set_fd(ssl, fd) } != 1 {
            unsafe { SSL_free(ssl); SSL_CTX_free(ctx); }
            return Err(crate::Error::Tls("SSL_set_fd failed".into()));
        }

        // SNI
        let host_cstr = CString::new(host)
            .map_err(|_| crate::Error::Tls("invalid hostname".into()))?;
        unsafe { SSL_ctrl(ssl, 55, 0, host_cstr.as_ptr() as *mut c_void); }

        let mut stream = Self { tcp, ssl, ctx };
        stream.async_handshake().await?;
        Ok(stream)
    }

    pub fn raw_fd(&self) -> RawFd {
        self.tcp.raw_fd()
    }

    async fn async_handshake(&mut self) -> crate::Result<()> {
        loop {
            let ret = unsafe { SSL_connect(self.ssl) };
            if ret == 1 { return Ok(()); }
            match unsafe { SSL_get_error(self.ssl, ret) } {
                SSL_ERROR_WANT_READ => wait_readable(self.tcp.raw_fd()).await?,
                SSL_ERROR_WANT_WRITE => wait_writable(self.tcp.raw_fd()).await?,
                err => return Err(crate::Error::Tls(format!("SSL_connect failed: error {}", err))),
            }
        }
    }

    pub async fn read(&mut self, buf: &mut [u8]) -> crate::Result<usize> {
        loop {
            let ret = unsafe { SSL_read(self.ssl, buf.as_mut_ptr() as *mut c_void, buf.len() as i32) };
            if ret > 0 { return Ok(ret as usize); }
            if ret == 0 { return Ok(0); } // EOF
            match unsafe { SSL_get_error(self.ssl, ret) } {
                SSL_ERROR_WANT_READ => wait_readable(self.tcp.raw_fd()).await?,
                SSL_ERROR_WANT_WRITE => wait_writable(self.tcp.raw_fd()).await?,
                err => return Err(crate::Error::Tls(format!("SSL_read failed: error {}", err))),
            }
        }
    }

    pub async fn write_all(&mut self, data: &[u8]) -> crate::Result<()> {
        let mut written = 0;
        while written < data.len() {
            let ret = unsafe {
                SSL_write(self.ssl, data[written..].as_ptr() as *const c_void, (data.len() - written) as i32)
            };
            if ret > 0 { written += ret as usize; continue; }
            match unsafe { SSL_get_error(self.ssl, ret) } {
                SSL_ERROR_WANT_READ => wait_readable(self.tcp.raw_fd()).await?,
                SSL_ERROR_WANT_WRITE => wait_writable(self.tcp.raw_fd()).await?,
                err => return Err(crate::Error::Tls(format!("SSL_write failed: error {}", err))),
            }
        }
        Ok(())
    }
}

impl Drop for AsyncTlsStream {
    fn drop(&mut self) {
        unsafe {
            SSL_shutdown(self.ssl);
            SSL_free(self.ssl);
            SSL_CTX_free(self.ctx);
        }
    }
}

// ── Reactor helpers ──────────────────────────────────────────────────────────

pub async fn wait_readable(fd: RawFd) -> crate::Result<()> {
    WaitFd { fd, writable: false, token: None }.await
}

pub async fn wait_writable(fd: RawFd) -> crate::Result<()> {
    WaitFd { fd, writable: true, token: None }.await
}

struct WaitFd {
    fd: RawFd,
    writable: bool,
    token: Option<u64>,
}

impl Future for WaitFd {
    type Output = crate::Result<()>;

    fn poll(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
        // 已经被唤醒过一次就 Ready
        if self.token.is_some() {
            self.token = None;
            return Poll::Ready(Ok(()));
        }

        let mut r = reactor().lock().unwrap();
        let t = if self.writable {
            r.register_writable(self.fd, cx.waker().clone())
        } else {
            r.register_readable(self.fd, cx.waker().clone())
        };
        self.token = Some(t);
        Poll::Pending
    }
}

impl Drop for WaitFd {
    fn drop(&mut self) {
        if let Some(t) = self.token.take() {
            reactor().lock().unwrap().remove(t);
        }
    }
}
```

```rust
#[derive(Debug, Clone, PartialEq)]
pub enum WsOpcode {
    Text,     // 0x1
    Binary,   // 0x2
    Close,    // 0x8
    Ping,     // 0x9
    Pong,     // 0xA
}

impl WsOpcode {
    fn to_byte(&self) -> u8 {
        match self {
            WsOpcode::Text => 0x1,
            WsOpcode::Binary => 0x2,
            WsOpcode::Close => 0x8,
            WsOpcode::Ping => 0x9,
            WsOpcode::Pong => 0xA,
        }
    }

    fn from_byte(b: u8) -> Option<Self> {
        match b {
            0x1 => Some(WsOpcode::Text),
            0x2 => Some(WsOpcode::Binary),
            0x8 => Some(WsOpcode::Close),
            0x9 => Some(WsOpcode::Ping),
            0xA => Some(WsOpcode::Pong),
            _ => None,
        }
    }
}

pub struct WsFrame {
    pub opcode: WsOpcode,
    pub payload: Vec<u8>,
}

impl WsFrame {
    pub fn text(s: &str) -> Self {
        WsFrame { opcode: WsOpcode::Text, payload: s.as_bytes().to_vec() }
    }

    pub fn close() -> Self {
        WsFrame { opcode: WsOpcode::Close, payload: vec![] }
    }

    pub fn pong(data: &[u8]) -> Self {
        WsFrame { opcode: WsOpcode::Pong, payload: data.to_vec() }
    }

    /// 编码为客户端帧（带 mask）
    pub fn encode(&self) -> Vec<u8> {
        let mut buf = Vec::new();
        // FIN + opcode
        buf.push(0x80 | self.opcode.to_byte());

        let len = self.payload.len();
        // 长度 + MASK bit
        if len < 126 {
            buf.push(0x80 | len as u8);
        } else if len <= 65535 {
            buf.push(0x80 | 126);
            buf.extend_from_slice(&(len as u16).to_be_bytes());
        } else {
            buf.push(0x80 | 127);
            buf.extend_from_slice(&(len as u64).to_be_bytes());
        }

        // Mask key（简单起见用固定种子，实际可用随机）
        let mask: [u8; 4] = generate_mask_key();
        buf.extend_from_slice(&mask);

        // Masked payload
        for (i, b) in self.payload.iter().enumerate() {
            buf.push(b ^ mask[i % 4]);
        }

        buf
    }

    /// 解码服务端帧（无 mask）。返回 (frame, consumed_bytes) 或 None（数据不完整）
    pub fn decode(data: &[u8]) -> crate::Result<Option<(Self, usize)>> {
        if data.len() < 2 { return Ok(None); }

        let opcode_byte = data[0] & 0x0F;
        let opcode = WsOpcode::from_byte(opcode_byte)
            .ok_or(crate::Error::Http(format!("unknown WS opcode: {}", opcode_byte)))?;
        let masked = data[1] & 0x80 != 0;
        let len_byte = data[1] & 0x7F;

        let (payload_len, header_end) = if len_byte < 126 {
            (len_byte as usize, 2)
        } else if len_byte == 126 {
            if data.len() < 4 { return Ok(None); }
            (u16::from_be_bytes([data[2], data[3]]) as usize, 4)
        } else {
            if data.len() < 10 { return Ok(None); }
            let len = u64::from_be_bytes([
                data[2], data[3], data[4], data[5],
                data[6], data[7], data[8], data[9],
            ]) as usize;
            (len, 10)
        };

        let mask_len = if masked { 4 } else { 0 };
        let total = header_end + mask_len + payload_len;
        if data.len() < total { return Ok(None); }

        let mut payload = data[header_end + mask_len..total].to_vec();
        if masked {
            let mask = &data[header_end..header_end + 4];
            for (i, b) in payload.iter_mut().enumerate() {
                *b ^= mask[i % 4];
            }
        }

        Ok(Some((WsFrame { opcode, payload }, total)))
    }
}

fn generate_mask_key() -> [u8; 4] {
    // 简易伪随机：基于时间戳的低位
    let t = std::time::SystemTime::now()
        .duration_since(std::time::UNIX_EPOCH)
        .unwrap_or_default()
        .as_nanos() as u64;
    let bytes = t.to_le_bytes();
    [bytes[0], bytes[1], bytes[2], bytes[3]]
}

/// WebSocket 握手（在已有的 stream 上）
pub fn build_upgrade_request(host: &str, path: &str) -> Vec<u8> {
    // Sec-WebSocket-Key 必须是 16 字节 base64
    let key = base64_encode(&generate_mask_key().repeat(4)[..16]);
    format!(
        "GET {} HTTP/1.1\r\n\
         Host: {}\r\n\
         Upgrade: websocket\r\n\
         Connection: Upgrade\r\n\
         Sec-WebSocket-Version: 13\r\n\
         Sec-WebSocket-Key: {}\r\n\
         \r\n",
        path, host, key
    ).into_bytes()
}

fn base64_encode(data: &[u8]) -> String {
    const TABLE: &[u8] = b"ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
    let mut result = String::new();
    for chunk in data.chunks(3) {
        let b0 = chunk[0] as u32;
        let b1 = if chunk.len() > 1 { chunk[1] as u32 } else { 0 };
        let b2 = if chunk.len() > 2 { chunk[2] as u32 } else { 0 };
        let triple = (b0 << 16) | (b1 << 8) | b2;
        result.push(TABLE[((triple >> 18) & 0x3F) as usize] as char);
        result.push(TABLE[((triple >> 12) & 0x3F) as usize] as char);
        if chunk.len() > 1 {
            result.push(TABLE[((triple >> 6) & 0x3F) as usize] as char);
        } else {
            result.push('=');
        }
        if chunk.len() > 2 {
            result.push(TABLE[(triple & 0x3F) as usize] as char);
        } else {
            result.push('=');
        }
    }
    result
}
```

```rust
use crate::core::json::JsonValue;

pub struct McpConfig {
    pub servers: Vec<(String, ServerConfig)>,
}

pub enum ServerConfig {
    Stdio { command: String, args: Vec<String>, env: Vec<(String, String)> },
    Sse { url: String },
    Http { url: String },
    WebSocket { url: String },
}

impl McpConfig {
    pub fn load(path: &str) -> crate::Result<Self> {
        match std::fs::read_to_string(path) {
            Ok(content) => Self::parse(&content),
            Err(_) => Ok(McpConfig { servers: vec![] }),
        }
    }

    pub fn parse(json_str: &str) -> crate::Result<Self> {
        let json = JsonValue::parse(json_str)?;
        let servers_obj = match json.get("mcpServers") {
            Some(obj) => obj,
            None => return Ok(McpConfig { servers: vec![] }),
        };

        let pairs = match servers_obj.as_object() {
            Some(p) => p,
            None => return Ok(McpConfig { servers: vec![] }),
        };

        let mut servers = Vec::new();
        for (name, cfg) in pairs {
            let typ = cfg.get("type").and_then(|v| v.as_str()).unwrap_or("stdio");
            let server_config = match typ {
                "stdio" => {
                    let command = cfg.get("command").and_then(|v| v.as_str())
                        .ok_or(crate::Error::Json(format!("MCP server '{}': missing command", name)))?
                        .to_string();
                    let args = cfg.get("args").and_then(|v| v.as_array())
                        .map(|arr| arr.iter().filter_map(|v| v.as_str().map(|s| s.to_string())).collect())
                        .unwrap_or_default();
                    let env = cfg.get("env").and_then(|v| v.as_object())
                        .map(|pairs| pairs.iter().filter_map(|(k, v)| {
                            v.as_str().map(|s| (k.clone(), s.to_string()))
                        }).collect())
                        .unwrap_or_default();
                    ServerConfig::Stdio { command, args, env }
                }
                "sse" => {
                    let url = cfg.get("url").and_then(|v| v.as_str())
                        .ok_or(crate::Error::Json(format!("MCP server '{}': missing url", name)))?
                        .to_string();
                    ServerConfig::Sse { url }
                }
                "http" => {
                    let url = cfg.get("url").and_then(|v| v.as_str())
                        .ok_or(crate::Error::Json(format!("MCP server '{}': missing url", name)))?
                        .to_string();
                    ServerConfig::Http { url }
                }
                "websocket" => {
                    let url = cfg.get("url").and_then(|v| v.as_str())
                        .ok_or(crate::Error::Json(format!("MCP server '{}': missing url", name)))?
                        .to_string();
                    ServerConfig::WebSocket { url }
                }
                other => return Err(crate::Error::Json(format!("MCP server '{}': unknown type '{}'", name, other))),
            };
            servers.push((name.clone(), server_config));
        }

        Ok(McpConfig { servers })
    }
}
```

```rust
    unsafe {
        let flags = fcntl(fd, 3); // F_GETFL
        fcntl(fd, 4, flags | 0o4000); // F_SETFL | O_NONBLOCK
    }
}

pub struct StdioTransport {
    child: Child,
    stdin: BufWriter<ChildStdin>,
    stdout_fd: RawFd,
    read_buf: Vec<u8>,
}

impl StdioTransport {
    pub fn spawn(
        command: &str,
        args: &[String],
        env: &[(String, String)],
    ) -> crate::Result<Self> {
        let mut cmd = Command::new(command);
        cmd.args(args)
            .stdin(Stdio::piped())
            .stdout(Stdio::piped())
            .stderr(Stdio::piped());
        for (k, v) in env {
            cmd.env(k, v);
        }

        let mut child = cmd.spawn()
            .map_err(|e| crate::Error::Mcp {
                server: command.to_string(),
                message: format!("spawn failed: {}", e),
            })?;

        let stdin = child.stdin.take()
            .ok_or(crate::Error::Mcp { server: command.into(), message: "no stdin".into() })?;
        let stdout = child.stdout.as_ref()
            .ok_or(crate::Error::Mcp { server: command.into(), message: "no stdout".into() })?;
        let stdout_fd = stdout.as_raw_fd();
        set_nonblocking(stdout_fd);

        Ok(StdioTransport {
            child,
            stdin: BufWriter::new(stdin),
            stdout_fd,
            read_buf: Vec::with_capacity(4096),
        })
    }
}

impl Transport for StdioTransport {
    fn send(&mut self, msg: JsonValue) -> Pin<Box<dyn Future<Output = crate::Result<()>> + Send + '_>> {
        Box::pin(async move {
            let line = format!("{}\n", msg);
            self.stdin.write_all(line.as_bytes())
                .map_err(|e| crate::Error::Io(e))?;
            self.stdin.flush()
                .map_err(|e| crate::Error::Io(e))?;
            Ok(())
        })
    }

    fn recv(&mut self) -> Pin<Box<dyn Future<Output = crate::Result<JsonValue>> + Send + '_>> {
        Box::pin(async move {
            loop {
                // 检查是否有完整行
                if let Some(pos) = self.read_buf.iter().position(|&b| b == b'\n') {
                    let line: Vec<u8> = self.read_buf.drain(..=pos).collect();
                    let s = std::str::from_utf8(&line)
                        .map_err(|_| crate::Error::Json("invalid UTF-8 from MCP server".into()))?;
                    return JsonValue::parse(s.trim());
                }

                // 等待 stdout 可读
                wait_readable(self.stdout_fd).await?;

                // 非阻塞读
                let mut buf = [0u8; 4096];
                let n = unsafe { read(self.stdout_fd, buf.as_mut_ptr(), buf.len()) };
                if n <= 0 {
                    return Err(crate::Error::Mcp {
                        server: String::new(),
                        message: "server process closed stdout".into(),
                    });
                }
                self.read_buf.extend_from_slice(&buf[..n as usize]);
            }
        })
    }

    fn close(&mut self) -> Pin<Box<dyn Future<Output = crate::Result<()>> + Send + '_>> {
        Box::pin(async move {
            // 关闭 stdin 管道（通知子进程 EOF）
            // BufWriter 的 inner ChildStdin 在 drop 时关闭 fd
            let _ = self.stdin.flush();
            // child.stdin 已被 take，所以这里 kill + wait
            let _ = self.child.kill();
            let _ = self.child.wait();
            Ok(())
        })
    }
}
```

```rust
use crate::core::json::{JsonValue, Number};
use crate::core::jsonrpc::{Request, Response, Notification, Message};
use crate::mcp::types::*;
use crate::mcp::transport::Transport;

pub struct McpClient<T: Transport> {
    transport: T,
    next_id: i64,
    pub server_capabilities: Option<ServerCapabilities>,
    pending_notifications: Vec<Notification>,
}

impl<T: Transport> McpClient<T> {
    pub fn new(transport: T) -> Self {
        McpClient {
            transport,
            next_id: 0,
            server_capabilities: None,
            pending_notifications: Vec::new(),
        }
    }

    async fn request(&mut self, method: &str, params: Option<JsonValue>) -> crate::Result<JsonValue> {
        let id = self.next_id;
        self.next_id += 1;
        let req = Request { id, method: method.into(), params };
        self.transport.send(req.to_json()).await?;

        loop {
            let msg_json = self.transport.recv().await?;
            let msg = Message::parse(&msg_json)?;
            match msg {
                Message::Response(Response::Result { id: rid, result }) if rid == id => {
                    return Ok(result);
                }
                Message::Response(Response::Error { id: rid, error }) if rid == id => {
                    return Err(crate::Error::JsonRpc { code: error.code, message: error.message });
                }
                Message::Notification(n) => {
                    self.pending_notifications.push(n);
                }
                _ => {} // 忽略不匹配的响应
            }
        }
    }

    async fn notify(&mut self, method: &str, params: Option<JsonValue>) -> crate::Result<()> {
        let notif = Notification { method: method.into(), params };
        self.transport.send(notif.to_json()).await
    }

    // ── Lifecycle ────────────────────────────────────────────────────────────

    pub async fn initialize(&mut self) -> crate::Result<ServerCapabilities> {
        let params = JsonValue::Object(vec![
            ("protocolVersion".into(), JsonValue::Str("2025-11-25".into())),
            ("capabilities".into(), JsonValue::Object(vec![])),
            ("clientInfo".into(), JsonValue::Object(vec![
                ("name".into(), JsonValue::Str("viv".into())),
                ("version".into(), JsonValue::Str("0.1.0".into())),
            ])),
        ]);
        let result = self.request("initialize", Some(params)).await?;
        let caps = ServerCapabilities::from_json(&result)?;
        self.server_capabilities = Some(caps);
        self.notify("notifications/initialized", None).await?;

        // 返回克隆不方便（没 Clone），重新解析
        ServerCapabilities::from_json(&result)
    }

    pub async fn shutdown(&mut self) -> crate::Result<()> {
        self.transport.close().await
    }

    // ── Tools ────────────────────────────────────────────────────────────────

    pub async fn list_tools(&mut self) -> crate::Result<Vec<McpTool>> {
        let mut all = Vec::new();
        let mut cursor: Option<String> = None;
        loop {
            let params = cursor.as_ref().map(|c| {
                JsonValue::Object(vec![("cursor".into(), JsonValue::Str(c.clone()))])
            });
            let result = self.request("tools/list", params).await?;
            if let Some(arr) = result.get("tools").and_then(|v| v.as_array()) {
                for t in arr {
                    all.push(McpTool::from_json(t)?);
                }
            }
            match result.get("nextCursor").and_then(|v| v.as_str()) {
                Some(c) => cursor = Some(c.to_string()),
                None => break,
            }
        }
        Ok(all)
    }

    pub async fn call_tool(&mut self, name: &str, args: &JsonValue) -> crate::Result<ToolCallResult> {
        let params = JsonValue::Object(vec![
            ("name".into(), JsonValue::Str(name.into())),
            ("arguments".into(), args.clone()),
        ]);
        let result = self.request("tools/call", Some(params)).await?;
        ToolCallResult::from_json(&result)
    }

    // ── Resources ────────────────────────────────────────────────────────────

    pub async fn list_resources(&mut self) -> crate::Result<Vec<McpResource>> {
        let result = self.request("resources/list", None).await?;
        McpResource::parse_list(&result)
    }

    pub async fn read_resource(&mut self, uri: &str) -> crate::Result<ResourceContent> {
        let params = JsonValue::Object(vec![
            ("uri".into(), JsonValue::Str(uri.into())),
        ]);
        let result = self.request("resources/read", Some(params)).await?;
        ResourceContent::from_json(&result)
    }

    // ── Prompts ──────────────────────────────────────────────────────────────

    pub async fn list_prompts(&mut self) -> crate::Result<Vec<McpPrompt>> {
        let result = self.request("prompts/list", None).await?;
        McpPrompt::parse_list(&result)
    }

    pub async fn get_prompt(&mut self, name: &str, args: &JsonValue) -> crate::Result<PromptMessages> {
        let params = JsonValue::Object(vec![
            ("name".into(), JsonValue::Str(name.into())),
            ("arguments".into(), args.clone()),
        ]);
        let result = self.request("prompts/get", Some(params)).await?;
        PromptMessages::from_json(&result)
    }
}
```

```rust
        } else { vec![] };
```

```rust
        JsonValue::parse(r#"{"type":"object","properties":{"server":{"type":"string"},"name":{"type":"string"},"arguments":{"type":"object"}},"required":["server","name"]}"#).unwrap()
    }
    fn permission_level(&self) -> PermissionLevel { PermissionLevel::ReadOnly }

    fn execute(&self, input: &JsonValue) -> Pin<Box<dyn Future<Output = crate::Result<String>> + Send + '_>> {
        let server = input.get("server").and_then(|v| v.as_str()).unwrap_or("").to_string();
        let name = input.get("name").and_then(|v| v.as_str()).unwrap_or("").to_string();
        let args = input.get("arguments").cloned().unwrap_or(JsonValue::Object(vec![]));
        Box::pin(async move {
            let mut mgr = self.manager.lock().unwrap();
            let result = mgr.get_prompt(&server, &name, &args).await?;
            let mut text = String::new();
            for msg in &result.messages {
                text.push_str(&format!("[{}] ", msg.role));
                match &msg.content {
                    super::types::ContentItem::Text(t) => text.push_str(t),
                    _ => text.push_str("(non-text content)"),
                }
                text.push('\n');
            }
            Ok(text)
        })
    }
}
```
