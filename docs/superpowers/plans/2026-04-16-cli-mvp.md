```rust
pub fn apply_raw_flags(termios: &mut Termios) {
    termios.c_iflag &= !(IXON | ICRNL);
    termios.c_oflag &= !(OPOST);
    termios.c_lflag &= !(ECHO | ICANON | ISIG | IEXTEN);
}

impl RawMode {
    pub fn enable(fd: RawFd) -> Result<Self, String> {
        let mut original = unsafe { std::mem::zeroed::<Termios>() };
        if unsafe { tcgetattr(fd, &mut original) } != 0 {
            return Err("tcgetattr failed".into());
        }
        let mut raw = original.clone();
        apply_raw_flags(&mut raw);
        // Set VMIN=1, VTIME=0 for blocking read of at least 1 byte
        raw.c_cc[6] = 1;  // VMIN
        raw.c_cc[5] = 0;  // VTIME
        if unsafe { tcsetattr(fd, TCSAFLUSH, &raw) } != 0 {
            return Err("tcsetattr failed".into());
        }
        Ok(RawMode { fd, original })
    }
}

impl Drop for RawMode {
    fn drop(&mut self) {
        unsafe {
            tcsetattr(self.fd, TCSAFLUSH, &self.original);
        }
    }
}
```

```rust
#[derive(Debug, Clone, Copy)]
pub enum Color {
    Black = 30,
    Red = 31,
    Green = 32,
    Yellow = 33,
    Blue = 34,
    Magenta = 35,
    Cyan = 36,
    White = 37,
}

impl AnsiWriter {
    pub fn new() -> Self {
        Self { buf: Vec::new() }
    }

    pub fn take(&mut self) -> Vec<u8> {
        std::mem::take(&mut self.buf)
    }

    pub fn move_to(&mut self, row: u16, col: u16) {
        use std::fmt::Write;
        write!(self, "\x1b[{};{}H", row + 1, col + 1).unwrap();
    }

    pub fn clear_screen(&mut self) {
        self.buf.extend_from_slice(b"\x1b[2J");
    }

    pub fn clear_line(&mut self) {
        self.buf.extend_from_slice(b"\x1b[2K");
    }

    pub fn fg_color(&mut self, color: Color) {
        use std::fmt::Write;
        write!(self, "\x1b[{}m", color as u8).unwrap();
    }

    pub fn bold(&mut self) {
        self.buf.extend_from_slice(b"\x1b[1m");
    }

    pub fn reset_style(&mut self) {
        self.buf.extend_from_slice(b"\x1b[0m");
    }

    pub fn hide_cursor(&mut self) {
        self.buf.extend_from_slice(b"\x1b[?25l");
    }

    pub fn show_cursor(&mut self) {
        self.buf.extend_from_slice(b"\x1b[?25h");
    }

    pub fn write_str(&mut self, s: &str) {
        self.buf.extend_from_slice(s.as_bytes());
    }

    pub fn write_bytes(&mut self, b: &[u8]) {
        self.buf.extend_from_slice(b);
    }
}

impl std::fmt::Write for AnsiWriter {
    fn write_str(&mut self, s: &str) -> std::fmt::Result {
        self.buf.extend_from_slice(s.as_bytes());
        Ok(())
    }
}
```

```rust
impl InputParser {
    pub fn new() -> Self {
        Self { buf: Vec::new() }
    }

    pub fn feed(&mut self, data: &[u8]) {
        self.buf.extend_from_slice(data);
    }

    pub fn next_event(&mut self) -> Option<KeyEvent> {
        if self.buf.is_empty() {
            return None;
        }

        let b = self.buf[0];

        // Control characters
        match b {
            3 => { self.buf.remove(0); return Some(KeyEvent::CtrlC); }
            4 => { self.buf.remove(0); return Some(KeyEvent::CtrlD); }
            13 => { self.buf.remove(0); return Some(KeyEvent::Enter); }
            127 => { self.buf.remove(0); return Some(KeyEvent::Backspace); }
            // Escape sequences
            0x1b => return self.parse_escape(),
            // UTF-8 multibyte
            0x80..=0xFF => return self.parse_utf8(),
            // Printable ASCII
            32..=126 => {
                self.buf.remove(0);
                return Some(KeyEvent::Char(b as char));
            }
            _ => {
                self.buf.remove(0);
                return Some(KeyEvent::Unknown(vec![b]));
            }
        }
    }

    fn parse_escape(&mut self) -> Option<KeyEvent> {
        if self.buf.len() < 2 {
            self.buf.remove(0);
            return Some(KeyEvent::Escape);
        }
        if self.buf[1] != b'[' {
            self.buf.remove(0);
            return Some(KeyEvent::Escape);
        }
        if self.buf.len() < 3 {
            // Incomplete sequence, treat as escape
            self.buf.remove(0);
            return Some(KeyEvent::Escape);
        }

        let event = match self.buf[2] {
            b'A' => { self.buf.drain(..3); Some(KeyEvent::Up) }
            b'B' => { self.buf.drain(..3); Some(KeyEvent::Down) }
            b'C' => { self.buf.drain(..3); Some(KeyEvent::Right) }
            b'D' => { self.buf.drain(..3); Some(KeyEvent::Left) }
            b'H' => { self.buf.drain(..3); Some(KeyEvent::Home) }
            b'F' => { self.buf.drain(..3); Some(KeyEvent::End) }
            b'3' if self.buf.len() >= 4 && self.buf[3] == b'~' => {
                self.buf.drain(..4);
                Some(KeyEvent::Delete)
            }
            _ => {
                // Unknown escape sequence, consume ESC[X
                let unknown = self.buf.drain(..3).collect();
                Some(KeyEvent::Unknown(unknown))
            }
        };
        event
    }

    fn parse_utf8(&mut self) -> Option<KeyEvent> {
        let first = self.buf[0];
        let len = if first & 0xE0 == 0xC0 { 2 }
                  else if first & 0xF0 == 0xE0 { 3 }
                  else if first & 0xF8 == 0xF0 { 4 }
                  else { 1 };

        if self.buf.len() < len {
            return None; // Incomplete UTF-8
        }

        let bytes: Vec<u8> = self.buf.drain(..len).collect();
        match std::str::from_utf8(&bytes) {
            Ok(s) => s.chars().next().map(KeyEvent::Char),
            Err(_) => Some(KeyEvent::Unknown(bytes)),
        }
    }
}
```

```rust
impl Screen {
    pub fn new(width: u16, height: u16) -> Self {
        let size = width as usize * height as usize;
        Screen {
            width,
            height,
            front: vec![Cell::default(); size],
            back: vec![Cell::default(); size],
        }
    }

    fn idx(&self, row: u16, col: u16) -> usize {
        row as usize * self.width as usize + col as usize
    }

    pub fn get(&self, row: u16, col: u16) -> Cell {
        self.back[self.idx(row, col)]
    }

    pub fn put(&mut self, row: u16, col: u16, ch: char) {
        let i = self.idx(row, col);
        self.back[i].ch = ch;
    }

    pub fn put_styled(&mut self, row: u16, col: u16, ch: char, fg: Option<u8>, bold: bool) {
        let i = self.idx(row, col);
        self.back[i] = Cell { ch, fg, bold };
    }

    pub fn put_str(&mut self, row: u16, col: u16, s: &str) {
        for (i, ch) in s.chars().enumerate() {
            let c = col + i as u16;
            if c < self.width {
                self.put(row, c, ch);
            }
        }
    }

    pub fn clear_back(&mut self) {
        self.back.fill(Cell::default());
    }

    /// Compare back vs front, return ANSI bytes to update the terminal, then sync front = back.
    pub fn diff(&mut self) -> Vec<u8> {
        use crate::terminal::output::AnsiWriter;

        let mut w = AnsiWriter::new();
        for row in 0..self.height {
            for col in 0..self.width {
                let i = self.idx(row, col);
                if self.back[i] != self.front[i] {
                    w.move_to(row, col);
                    let cell = &self.back[i];
                    if cell.bold {
                        w.bold();
                    }
                    if let Some(fg) = cell.fg {
                        use std::fmt::Write;
                        write!(w, "\x1b[{}m", fg).unwrap();
                    }
                    let mut buf = [0u8; 4];
                    let s = cell.ch.encode_utf8(&mut buf);
                    w.write_str(s);
                    if cell.bold || cell.fg.is_some() {
                        w.reset_style();
                    }
                }
            }
        }
        // Sync front to back
        self.front.copy_from_slice(&self.back);
        w.take()
    }
}
```

```rust
impl TlsStream {
    pub fn connect(host: &str, port: u16) -> Result<Self, String> {
        unsafe {
            OPENSSL_init_ssl(0, std::ptr::null());

            let method = TLS_client_method();
            let ctx = SSL_CTX_new(method);
            if ctx.is_null() {
                return Err("SSL_CTX_new failed".into());
            }

            SSL_CTX_set_default_verify_paths(ctx);
            SSL_CTX_set_verify(ctx, SSL_VERIFY_PEER, std::ptr::null());

            let ssl = SSL_new(ctx);
            if ssl.is_null() {
                SSL_CTX_free(ctx);
                return Err("SSL_new failed".into());
            }

            let tcp = super::tcp::connect(host, port)?;
            SSL_set_fd(ssl, tcp.as_raw_fd());

            // SNI
            let host_cstr = format!("{}\0", host);
            SSL_set_tlsext_host_name(ssl, host_cstr.as_ptr());

            let ret = SSL_connect(ssl);
            if ret != 1 {
                let err = SSL_get_error(ssl, ret);
                SSL_free(ssl);
                SSL_CTX_free(ctx);
                return Err(format!("SSL_connect failed: error {}", err));
            }

            Ok(TlsStream { ssl, ctx, _tcp: tcp })
        }
    }
}

impl Read for TlsStream {
    fn read(&mut self, buf: &mut [u8]) -> std::io::Result<usize> {
        let ret = unsafe { SSL_read(self.ssl, buf.as_mut_ptr(), buf.len() as i32) };
        if ret > 0 {
            Ok(ret as usize)
        } else {
            Err(std::io::Error::new(std::io::ErrorKind::Other, "SSL_read failed"))
        }
    }
}

impl Write for TlsStream {
    fn write(&mut self, buf: &[u8]) -> std::io::Result<usize> {
        let ret = unsafe { SSL_write(self.ssl, buf.as_ptr(), buf.len() as i32) };
        if ret > 0 {
            Ok(ret as usize)
        } else {
            Err(std::io::Error::new(std::io::ErrorKind::Other, "SSL_write failed"))
        }
    }

    fn flush(&mut self) -> std::io::Result<()> {
        Ok(())
    }
}

impl Drop for TlsStream {
    fn drop(&mut self) {
        unsafe {
            SSL_shutdown(self.ssl);
            SSL_free(self.ssl);
            SSL_CTX_free(self.ctx);
        }
    }
}
```

```rust
impl SseParser {
    pub fn new() -> Self {
        Self { buf: String::new() }
    }

    pub fn feed(&mut self, data: &str) {
        self.buf.push_str(data);
    }

    pub fn drain(&mut self) -> Vec<SseEvent> {
        let mut events = Vec::new();
        while let Some(pos) = self.buf.find("\n\n") {
            let block = self.buf[..pos].to_string();
            self.buf = self.buf[pos + 2..].to_string();

            let mut event = None;
            let mut data_parts = Vec::new();

            for line in block.lines() {
                if line.starts_with(':') {
                    continue; // comment
                }
                if let Some(value) = line.strip_prefix("event: ").or_else(|| line.strip_prefix("event:")) {
                    event = Some(value.trim().to_string());
                } else if let Some(value) = line.strip_prefix("data: ").or_else(|| line.strip_prefix("data:")) {
                    data_parts.push(value.trim().to_string());
                }
            }

            if !data_parts.is_empty() {
                events.push(SseEvent {
                    event,
                    data: data_parts.join("\n"),
                });
            }
        }
        events
    }
}
```

```rust
impl ApiClient {
    pub fn new(api_key: String, model: String) -> Self {
        Self { api_key, model }
    }

    pub fn from_env(model: String) -> Result<Self, String> {
        let api_key = std::env::var("ANTHROPIC_API_KEY")
            .map_err(|_| "ANTHROPIC_API_KEY environment variable not set".to_string())?;
        Ok(Self::new(api_key, model))
    }

    pub fn build_request(&self, messages: &[Message]) -> HttpRequest {
        let msgs: Vec<JsonValue> = messages
            .iter()
            .map(|m| {
                JsonValue::Object(vec![
                    ("role".into(), JsonValue::Str(m.role.clone())),
                    ("content".into(), JsonValue::Str(m.content.clone())),
                ])
            })
            .collect();

        let body = JsonValue::Object(vec![
            ("model".into(), JsonValue::Str(self.model.clone())),
            ("max_tokens".into(), JsonValue::Number(16000.0)),
            ("stream".into(), JsonValue::Bool(true)),
            ("messages".into(), JsonValue::Array(msgs)),
        ]);

        HttpRequest {
            method: "POST".into(),
            path: "/v1/messages".into(),
            headers: vec![
                ("Host".into(), "api.anthropic.com".into()),
                ("Content-Type".into(), "application/json".into()),
                ("x-api-key".into(), self.api_key.clone()),
                ("anthropic-version".into(), "2023-06-01".into()),
            ],
            body: Some(body.to_string()),
        }
    }

    /// Send a streaming request and call `on_text` for each text delta received.
    pub fn stream(
        &self,
        messages: &[Message],
        mut on_text: impl FnMut(&str),
    ) -> Result<String, String> {
        use crate::net::tls::TlsStream;
        use crate::net::sse::SseParser;
        use std::io::{Read, Write};

        let req = self.build_request(messages);
        let raw_req = req.to_bytes();

        let mut stream = TlsStream::connect("api.anthropic.com", 443)?;
        stream.write_all(&raw_req).map_err(|e| e.to_string())?;

        // Read response header first
        let mut header_buf = Vec::new();
        let mut byte = [0u8; 1];
        loop {
            stream.read(&mut byte).map_err(|e| e.to_string())?;
            header_buf.push(byte[0]);
            if header_buf.ends_with(b"\r\n\r\n") {
                break;
            }
        }

        // Parse status
        let header_str = String::from_utf8_lossy(&header_buf);
        let status_line = header_str.lines().next().unwrap_or("");
        let status: u16 = status_line
            .split_whitespace()
            .nth(1)
            .and_then(|s| s.parse().ok())
            .unwrap_or(0);

        if status != 200 {
            // Read error body
            let mut error_body = Vec::new();
            let _ = stream.read_to_end(&mut error_body);
            let error_text = String::from_utf8_lossy(&error_body);
            return Err(format!("API error {}: {}", status, error_text));
        }

        // Stream SSE events
        let mut sse = SseParser::new();
        let mut full_response = String::new();
        let mut read_buf = [0u8; 4096];

        loop {
            let n = stream.read(&mut read_buf).map_err(|e| e.to_string())?;
            if n == 0 {
                break;
            }
            let chunk = String::from_utf8_lossy(&read_buf[..n]);
            sse.feed(&chunk);

            for event in sse.drain() {
                if event.event.as_deref() == Some("message_stop") {
                    return Ok(full_response);
                }
                if let Some(text) = extract_delta_text(&event.data) {
                    full_response.push_str(&text);
                    on_text(&text);
                }
            }
        }

        Ok(full_response)
    }
}

pub fn extract_delta_text(data: &str) -> Option<String> {
    let val = JsonValue::parse(data).ok()?;
    let delta = val.get("delta")?;
    if delta.get("type").and_then(|v| v.as_str()) == Some("text_delta") {
        delta.get("text").and_then(|v| v.as_str()).map(|s| s.to_string())
    } else {
        None
    }
}
```

```rust
impl Epoll {
    pub fn new() -> Result<Self, String> {
        let fd = unsafe { epoll_create1(0) };
        if fd < 0 {
            return Err("epoll_create1 failed".into());
        }
        Ok(Self { fd })
    }

    pub fn add(&self, target_fd: RawFd, token: u64) -> Result<(), String> {
        let mut event = EpollEvent {
            events: EPOLLIN,
            data: token,
        };
        let ret = unsafe { epoll_ctl(self.fd, EPOLL_CTL_ADD, target_fd, &mut event) };
        if ret < 0 {
            return Err("epoll_ctl ADD failed".into());
        }
        Ok(())
    }

    pub fn remove(&self, target_fd: RawFd) -> Result<(), String> {
        let ret = unsafe { epoll_ctl(self.fd, EPOLL_CTL_DEL, target_fd, std::ptr::null_mut()) };
        if ret < 0 {
            return Err("epoll_ctl DEL failed".into());
        }
        Ok(())
    }

    pub fn wait(&self, timeout_ms: i32) -> Result<Vec<u64>, String> {
        let mut events = [EpollEvent { events: 0, data: 0 }; 16];
        let n = unsafe {
            epoll_wait(self.fd, events.as_mut_ptr(), events.len() as i32, timeout_ms)
        };
        if n < 0 {
            return Err("epoll_wait failed".into());
        }
        Ok((0..n as usize).map(|i| events[i].data).collect())
    }
}

impl Drop for Epoll {
    fn drop(&mut self) {
        unsafe { close(self.fd); }
    }
}
```

```rust
#[derive(Debug, PartialEq)]
pub enum EditAction {
    Continue,
    Submit(String),
    Exit,
    Interrupt,
}

impl LineEditor {
    pub fn new() -> Self {
        Self {
            buf: String::new(),
            cursor: 0,
        }
    }

    pub fn handle_key(&mut self, key: KeyEvent) -> EditAction {
        match key {
            KeyEvent::Char(ch) => {
                self.buf.insert(self.cursor, ch);
                self.cursor += ch.len_utf8();
                EditAction::Continue
            }
            KeyEvent::Enter => {
                let line = std::mem::take(&mut self.buf);
                self.cursor = 0;
                EditAction::Submit(line)
            }
            KeyEvent::Backspace => {
                if self.cursor > 0 {
                    // Find char boundary before cursor
                    let prev = self.buf[..self.cursor]
                        .char_indices()
                        .last()
                        .map(|(i, _)| i)
                        .unwrap_or(0);
                    self.buf.remove(prev);
                    self.cursor = prev;
                }
                EditAction::Continue
            }
            KeyEvent::Delete => {
                if self.cursor < self.buf.len() {
                    self.buf.remove(self.cursor);
                }
                EditAction::Continue
            }
            KeyEvent::Left => {
                if self.cursor > 0 {
                    self.cursor = self.buf[..self.cursor]
                        .char_indices()
                        .last()
                        .map(|(i, _)| i)
                        .unwrap_or(0);
                }
                EditAction::Continue
            }
            KeyEvent::Right => {
                if self.cursor < self.buf.len() {
                    self.cursor += self.buf[self.cursor..].chars().next().map(|c| c.len_utf8()).unwrap_or(0);
                }
                EditAction::Continue
            }
            KeyEvent::Home => {
                self.cursor = 0;
                EditAction::Continue
            }
            KeyEvent::End => {
                self.cursor = self.buf.len();
                EditAction::Continue
            }
            KeyEvent::CtrlC => EditAction::Interrupt,
            KeyEvent::CtrlD => {
                if self.buf.is_empty() {
                    EditAction::Exit
                } else {
                    EditAction::Continue
                }
            }
            _ => EditAction::Continue,
        }
    }
}
```

```rust
pub fn run() -> Result<(), String> {
    let api = ApiClient::from_env("claude-opus-4-6".into())?;
    let mut messages: Vec<Message> = Vec::new();
    let mut editor = LineEditor::new();
    let mut input_parser = InputParser::new();
    let mut out = stdout();

    // Print banner (raw mode needs \r\n)
    write!(out, "\x1b[1m\x1b[36mviv\x1b[0m — AI coding agent (type /exit or Ctrl+D to quit)\r\n\r\n").unwrap();
    out.flush().unwrap();

    let _raw = RawMode::enable(0)?;

    loop {
        // Prompt
        write!(out, "\x1b[1m\x1b[32m> \x1b[0m").unwrap();
        out.flush().unwrap();

        // Collect user input
        let user_input = match read_line(&mut editor, &mut input_parser, &mut out)? {
            Some(line) => line,
            None => return Ok(()), // EOF/exit
        };

        let trimmed = user_input.trim();
        if trimmed.is_empty() {
            continue;
        }
        if trimmed == "/exit" {
            write!(out, "Bye!\r\n").unwrap();
            out.flush().unwrap();
            return Ok(());
        }

        // Add user message
        messages.push(Message {
            role: "user".into(),
            content: user_input.clone(),
        });

        // Stream response
        write!(out, "\r\n\x1b[1m\x1b[35mclaude:\x1b[0m ").unwrap();
        out.flush().unwrap();

        let mut full_response = String::new();
        match api.stream(&messages, |text| {
            write!(out, "{}", text).unwrap();
            out.flush().unwrap();
            full_response.push_str(text);
        }) {
            Ok(response) => {
                messages.push(Message {
                    role: "assistant".into(),
                    content: response,
                });
            }
            Err(e) => {
                write!(out, "\r\n\x1b[31mError: {}\x1b[0m", e).unwrap();
            }
        }

        write!(out, "\r\n\r\n").unwrap();
        out.flush().unwrap();
    }
}

fn read_line(
    editor: &mut LineEditor,
    parser: &mut InputParser,
    out: &mut impl Write,
) -> Result<Option<String>, String> {
    let mut sin = stdin();
    loop {
        let mut buf = [0u8; 256];
        let n = sin.read(&mut buf).map_err(|e| e.to_string())?;
        if n == 0 {
            return Ok(None);
        }
        parser.feed(&buf[..n]);

        while let Some(key) = parser.next_event() {
            match editor.handle_key(key) {
                EditAction::Continue => {
                    write!(out, "\r\x1b[2K\x1b[1m\x1b[32m> \x1b[0m{}", editor.buf).unwrap();
                    // Move cursor to correct position
                    let visual_cursor = 2 + editor.buf[..editor.cursor].chars().count();
                    write!(out, "\r\x1b[{}C", visual_cursor).unwrap();
                    out.flush().unwrap();
                }
                EditAction::Submit(line) => {
                    write!(out, "\r\n").unwrap();
                    out.flush().unwrap();
                    return Ok(Some(line));
                }
                EditAction::Exit => {
                    write!(out, "\r\nBye!\r\n").unwrap();
                    out.flush().unwrap();
                    return Ok(None);

                EditAction::Interrupt => {
                    editor.buf.clear();
                    editor.cursor = 0;
                    write!(out, "^C\r\n\x1b[1m\x1b[32m> \x1b[0m").unwrap();
```
